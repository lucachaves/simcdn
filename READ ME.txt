        //HACK
        double UDP::ephemeralPortsUsage() {
	        double usage = (double)lastEphemeralPort - EPHEMERAL_PORTRANGE_START;
	        double total = EPHEMERAL_PORTRANGE_END - EPHEMERAL_PORTRANGE_START;
	        return usage / total;
        }
